// 응답 표시 함수 function displayResponse(response) { const chatMessages = document.getElementById('chatMessages'); const messageDiv = document.createElement('div'); messageDiv.className = 'message user-message'; if (response.success) { // 의도 분석 결과 표시 if (response.intent) { const intentInfo = document.createElement('div'); intentInfo.className = 'intent-info'; intentInfo.innerHTML = ` 🎯 ${response.intent.type === 'informational' ? '정보 요청' : '액션 요청'}: ${response.intent.reason} `; messageDiv.appendChild(intentInfo); } // 사용자 응답 표시 if (response.userResponse) { const userResponseDiv = document.createElement('div'); userResponseDiv.className = 'user-response'; if (response.userResponse.type === 'news_summary') { // 뉴스 요약 응답 userResponseDiv.innerHTML = `

${response.userResponse.title}

${response.userResponse.summary}

${response.userResponse.sections.map(section => `
${section.name}
${section.news.map(item => `
📰 ${item.title}
`).join('')}
`).join('')}
📊 총 ${response.userResponse.totalNews}개의 뉴스 | ${new Date(response.userResponse.timestamp).toLocaleString()} `; } else if (response.userResponse.type === 'action') { // 액션 응답 userResponseDiv.innerHTML = `

✅ ${response.userResponse.message}

🌐 브라우저 상태: ${response.userResponse.browserStatus} `; } else { // 일반 응답 userResponseDiv.innerHTML = `

${response.userResponse.message}

`; } messageDiv.appendChild(userResponseDiv); } // 실행 결과 요약 if (response.executionResult && response.executionResult.executionSummary) { const summaryDiv = document.createElement('div'); summaryDiv.className = 'execution-summary'; const summary = response.executionResult.executionSummary; summaryDiv.innerHTML = `
📊 실행 요약 (클릭하여 펼치기)

📋 총 ${summary.total}개 명령어 | ✅ 성공: ${summary.successful} | ❌ 실패: ${summary.failed}

${summary.commands.map(cmd => `
${cmd.step} ${cmd.action} ${cmd.description} ${cmd.status === 'success' ? '✅' : cmd.status === 'failed' ? '❌' : '⏭️'}
`).join('')}
`; messageDiv.appendChild(summaryDiv); } } else { // 오류 응답 messageDiv.innerHTML = `
❌ 오류가 발생했습니다: ${response.error}
`; } chatMessages.appendChild(messageDiv); chatMessages.scrollTop = chatMessages.scrollHeight; } .message { margin: 10px 0; padding: 15px; border-radius: 10px; max-width: 80%; word-wrap: break-word; } .user-message { background-color: #e3f2fd; margin-left: auto; text-align: right; } .bot-message { background-color: #f5f5f5; margin-right: auto; } .intent-info { background-color: #fff3e0; padding: 8px; border-radius: 5px; margin-bottom: 10px; text-align: left; } .user-response { text-align: left; margin: 10px 0; } .user-response h4 { color: #1976d2; margin: 0 0 10px 0; } .news-sections { margin: 15px 0; } .news-section { margin: 15px 0; padding: 10px; background-color: #fafafa; border-radius: 8px; border-left: 4px solid #2196f3; } .news-section h5 { margin: 0 0 10px 0; color: #1565c0; } .news-items { display: flex; flex-direction: column; gap: 8px; } .news-item { padding: 8px; background-color: white; border-radius: 5px; border: 1px solid #e0e0e0; } .news-link { color: #1976d2; text-decoration: none; display: block; } .news-link:hover { text-decoration: underline; } .execution-summary { margin-top: 15px; padding: 10px; background-color: #f8f9fa; border-radius: 8px; border: 1px solid #dee2e6; } .execution-summary summary { cursor: pointer; font-weight: bold; color: #495057; } .summary-details { margin-top: 10px; padding-top: 10px; border-top: 1px solid #dee2e6; } .command-list { margin-top: 10px; } .command-item { display: flex; gap: 10px; padding: 5px; margin: 2px 0; border-radius: 3px; font-size: 0.9em; } .command-item.success { background-color: #d4edda; } .command-item.failed { background-color: #f8d7da; } .command-item.skipped { background-color: #fff3cd; } .step { font-weight: bold; min-width: 30px; } .action { color: #6c757d; min-width: 80px; } .description { flex: 1; } .status { min-width: 20px; } .error-message { color: #dc3545; background-color: #f8d7da; padding: 10px; border-radius: 5px; border: 1px solid #f5c6cb; }
